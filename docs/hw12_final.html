Question 1

from math import exp
import json
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go

gap = px.data.gapminder().query("year==1972")

europe = gap.query("continent == 'Europe'") #new

fig = px.choropleth(
    europe,
    locations = "iso_alpha",
    color = "gdpPercap",
    hover_name = "country",
    hover_data = {
        "gdpPercap": ":,.0f",
        "pop": ":,",
        "iso_alpha": False
    },
    color_continuous_scale = "Viridis",
    title = "GDP By Country in Europe (1972)" 
)

fig.update_geos(
    scope = "europe",
    showland = True,
    landcolor = "rgb(240,240,240)"
)   

fig.show()

Something new

fig = px.choropleth(
    europe,
    locations = "iso_alpha",
    color = "gdpPercap",
    hover_name = "country",
    hover_data = {
        "gdpPercap": ":,.0f",
        "pop": ":,",
        "iso_alpha": False
    },
    color_continuous_scale = "Viridis",
    range_color = (0, 5000),
    title = "GDP By Country in Europe (1972)" 
)

fig.update_geos(
    scope = "europe",
    showland = True,
    landcolor = "rgb(240,240,240)"
)   
fig.show()

I used range_color to change the color range for GDP on the map

The color range is now less variable, and doesn’t show differences in GDP with color as well
